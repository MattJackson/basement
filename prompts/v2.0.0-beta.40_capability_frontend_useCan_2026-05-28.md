### 2. useCan hook

```ts
// useCan reads the user's capabilities from useUser and returns
// whether the active role grants the given capability.
//
// Usage:
//   const canEditCluster = useCan(Cap.ClusterWiringUpdate);
//   if (!canEditCluster) return null;
//
// For cluster-scoped capabilities, the backend enforces scoping
// via the JWT's activeRole.Cluster — the FE just hides UI based
// on whether the role grants the capability AT ALL. The user
// has only one active cluster scope at a time anyway.
import { useUser } from "./useUser";
import type { CapabilityKey } from "./capabilities";

export function useCan(capability: CapabilityKey): boolean {
  const { data } = useUser();
  return Boolean(data?.capabilities?.includes(capability));
}
```

### 3. Route capability map + ProtectedRoute rewrite

`route-capabilities.ts`:

```ts
import { Cap, type CapabilityKey } from "./capabilities";

// Maps route patterns to the capability required to access them.
// Patterns may contain `:cid` placeholders — they match any value.
// Order: longest specific pattern first.
export interface RouteRule {
  match: RegExp;
  capability: CapabilityKey;
}

export const ROUTE_RULES: RouteRule[] = [
  // Cluster wiring (UI Admin)
  { match: /^\/admin\/clusters$/,                  capability: Cap.ClusterWiringList },
  { match: /^\/admin\/clusters\/new$/,             capability: Cap.ClusterWiringCreate },
  { match: /^\/admin\/clusters\/[^/]+\/edit$/,     capability: Cap.ClusterWiringUpdate },

  // Cluster contents (Cluster Admin)
  { match: /^\/admin\/clusters\/[^/]+$/,           capability: Cap.ClusterContentsRead },
  { match: /^\/admin\/clusters\/[^/]+\/buckets($|\/)/, capability: Cap.ClusterContentsRead },
  { match: /^\/admin\/clusters\/[^/]+\/keys($|\/)/,    capability: Cap.ClusterContentsRead },

  // Cross-cluster aggregates (UI Admin)
  { match: /^\/admin\/buckets$/,                   capability: Cap.ClusterBucketsAggr },
  { match: /^\/admin\/usage$/,                     capability: Cap.ClusterUsageAggr },

  // Platform (UI Admin)
  { match: /^\/admin\/users($|\/)/,                capability: Cap.PlatformUsersList },
  { match: /^\/admin\/policies($|\/)/,             capability: Cap.PlatformPoliciesList },
  { match: /^\/admin\/service-accounts($|\/)/,     capability: Cap.PlatformServiceAccountsList },
  { match: /^\/admin\/audit$/,                     capability: Cap.PlatformAuditRead },
  { match: /^\/admin\/system$/,                    capability: Cap.PlatformSystemWrite },
  { match: /^\/admin\/skins($|\/)/,                capability: Cap.PlatformSkinsWrite },
  { match: /^\/admin\/oidc($|\/)/,                 capability: Cap.PlatformOIDCWrite },
  { match: /^\/admin\/onboarding($|\/)/,           capability: Cap.PlatformOnboardingWrite },
];

export function findRouteRule(pathname: string): RouteRule | undefined {
  return ROUTE_RULES.find((r) => r.match.test(pathname));
}
```

`ProtectedRoute.tsx` rewrite:

```tsx
import { useEffect } from "react";
import { useNavigate, useLocation } from "@tanstack/react-router";
import { toast } from "sonner";
import { useUser } from "./useUser";
import { findRouteRule } from "./route-capabilities";
import LoadingSpinner from "@/shared/ui/LoadingSpinner";
import type { ReactNode } from "react";

export function ProtectedRoute({ children }: { children: ReactNode }) {
  const navigate = useNavigate();
  const location = useLocation();
  const { data, isLoading } = useUser();

  useEffect(() => {
    if (isLoading) return;
    if (location.pathname === "/login") return;

    if (!data) {
      const next = location.pathname.startsWith("/admin") ? location.pathname : "/files";
      navigate({ to: "/login", search: { next } });
      return;
    }

    if (!location.pathname.startsWith("/admin")) return;

    const rule = findRouteRule(location.pathname);
    if (!rule) return; // Unmapped /admin/* route — fall through (defense in depth via BE)

    const caps = data.capabilities ?? [];
    if (!caps.includes(rule.capability)) {
      toast.info(`Switch to a role with ${roleNameFor(rule.capability)} to access this page.`);
      navigate({ to: "/files" });
    }
  }, [isLoading, data, navigate, location]);

  if (isLoading || !data) return <LoadingSpinner />;
  return <>{children}</>;
}

// roleNameFor returns the human role name that grants a capability,
// for the bounce toast. (Computed from capability prefix — keep
// simple. Localize via i18n in a follow-up.)
function roleNameFor(cap: string): string {
  if (cap.startsWith("platform.") || cap.startsWith("cluster.wiring.") ||
      cap.startsWith("cluster.buckets.aggregate") || cap.startsWith("cluster.usage.aggregate")) {
    return "UI Admin";
  }
  if (cap.startsWith("cluster.")) {
    return "Cluster Admin for this cluster";
  }
  return "an appropriate";
}
```
